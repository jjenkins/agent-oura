```javascript
// ouraGet: single authenticated GET. Throws typed errors:
//   'DATA_NOT_SYNCED' (err.message) when data:[] returned
//   'RATE_LIMITED' with err.retryAfter
//   'AUTH_EXPIRED', 'AUTH_FORBIDDEN', 'MEMBERSHIP_REQUIRED', 'APP_UPDATE_REQUIRED'
export async function ouraGet(path, params = {})

// ouraGetWithRetry: wraps ouraGet with retry logic. Converts typed errors
// to human-readable strings. NOT suitable for dashboard parallel fetch
// because it transforms DATA_NOT_SYNCED into a string message.
export async function ouraGetWithRetry(path, params = {}, attempt = 0)

// formatError: maps error code strings to user-friendly messages
export function formatError(error)
```

  <action>
Create `.claude/skills/oura/scripts/dashboard.mjs` as an ESM script. The script must:

**Imports and date:**
- `import { ouraGet } from './client.mjs';` -- use ouraGet directly, NOT ouraGetWithRetry, per Research Pitfall 1. ouraGet throws typed errors (err.message === 'DATA_NOT_SYNCED') that are easy to classify.
- Derive today's date as `const today = new Date().toLocaleDateString('en-CA');` -- this produces 'YYYY-MM-DD' in local timezone, per Research Pattern 2. Do NOT use `.toISOString().split('T')[0]` (UTC, causes off-by-one).

**Parallel fetch with Promise.allSettled:**
- Fetch all four daily endpoints in parallel using `Promise.allSettled()` (NOT `Promise.all()` -- per D-09, partial sync must show whatever is available):
  - `ouraGet('/usercollection/daily_readiness', { start_date: today, end_date: today })`
  - `ouraGet('/usercollection/daily_sleep', { start_date: today, end_date: today })`
  - `ouraGet('/usercollection/daily_activity', { start_date: today, end_date: today })`
  - `ouraGet('/usercollection/daily_stress', { start_date: today, end_date: today })`

**Record extraction function:**
- Create `extractRecord(result)` that:
  - Returns `null` if `result.status === 'rejected'` (DATA_NOT_SYNCED, MEMBERSHIP_REQUIRED, or any error)
  - Returns `null` if `result.value?.data` is not an array or is empty
  - Returns `data[0]` otherwise (today's single record)

**Global no-data check (D-11):**
- If all four extracted records are null, output exactly: `Today's data hasn't synced yet. Check your Oura app.\n` and `process.exit(0)`.

**Contributor sort function (D-05, D-06):**
- Create `sortedContributors(contributors)` that:
  - Returns `[]` if contributors is falsy
  - Filters out entries where value is `null` or `undefined`
  - Sorts by value ascending (lowest/worst first, per D-06)
  - Returns array of `[key, value]` pairs

**Section output (D-01, D-02, D-03, D-04, D-09, D-10):**
Build output lines starting with `=== Daily Health Dashboard ===` and `Date: ${today}`.

For each section, ONLY output if the extracted record is not null (D-10 -- omit entirely if not synced):

*Readiness section (D-02, D-03, D-05):*
- Header: `--- Readiness: ${readiness.score ?? 'pending'} ---`
- All contributors from `sortedContributors(readiness.contributors)` as `  ${key}: ${val}`
- If `readiness.temperature_deviation` is not null/undefined, append `  temperature_deviation: ${readiness.temperature_deviation} C`

*Sleep section (D-02, D-03, D-05):*
- Header: `--- Sleep: ${sleep.score ?? 'pending'} ---`
- All contributors from `sortedContributors(sleep.contributors)` as `  ${key}: ${val}`

*Activity section (D-02, D-03, D-08 -- no contributors):*
- Header: `--- Activity: ${activity.score ?? 'pending'} ---`
- Show 2-3 key metrics: `  active_calories: ${activity.active_calories}`, `  steps: ${activity.steps}`, `  equivalent_walking_distance: ${activity.equivalent_walking_distance} m` (if present)

*Stress section (D-02, D-04, D-08 -- no contributors, no numeric score):*
- Header: `--- Stress: ${stress.day_summary ?? 'pending'} ---` (use day_summary, NOT score -- stress has no score field, per Research Pitfall 3)
- Show `  stress_high: ${stress.stress_high}s` and `  recovery_high: ${stress.recovery_high}s` if not null

**Output:**
- Join all lines with `\n` and write to stdout via `process.stdout.write(lines.join('\n') + '\n');`
- Wrap the entire script body in a try/catch. On unexpected error, output the error message to stderr and exit with code 1.

**What NOT to do:**
- Do NOT import chalk or cli-table3 -- plain text output, Claude handles presentation
- Do NOT use `ouraGetWithRetry` -- it transforms error codes into string messages, making DATA_NOT_SYNCED indistinguishable from real errors
- Do NOT show empty placeholder sections for unsynced data (D-10)
- Do NOT fall back to yesterday's data (D-11)
- Do NOT hardcode human-readable contributor labels -- output raw API key names (D-07)
